    \n\

    \n:local mac \$leaseActMAC\

    \n  :if ([:len [/ip dhcp-server lease find where address=\$ip dynamic=no]] = 0) do={\
    \n    :if ([:len [/ip dns static find where comment=\$tag address=\$ip]] > 0) do={\
    \n      :return\
    \n    }\
    \n    :local hn \$\"lease-hostname\"\
    \n    :if ([:len \$hn] = 0) do={ :set hn \"host\" }\
    \n    :local clean (\$hn . \".\" . \$zone)\
    \n    :local sfx ([:pick \$mac 9 11] . [:pick \$mac 12 14] . [:pick \$mac 15 17])\
    \n    :local suffixed (\$hn . \"-\" . \$sfx . \".\" . \$zone)\
    \n    :foreach r in=[/ip dns static find where comment=\$tag address=\$ip] do={ /ip dns static remove \$r }\
    \n    :if ([:len [/ip dns static find where comment=\$tag name=\$clean]] = 0) do={\
    \n      /ip dns static add name=\$clean address=\$ip comment=\$tag ttl=5m\
    \n    } else={\
    \n      :foreach r in=[/ip dns static find where comment=\$tag name=\$suffixed] do={ /ip dns static remove \$r }\
    \n      /ip dns static add name=\$suffixed address=\$ip comment=\$tag ttl=5m\
    \n    }\
    \n  }\
    \n} else={\
    \n  :foreach r in=[/ip dns static find where comment=\$tag address=\$ip] do={ /ip dns static remove \$r }\
    \n}" lease-time=8h name=dhcp-vlan1
